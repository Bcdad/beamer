\section{Transformer、BERT 与知识蒸馏的学习}
 
\begin{frame}{Transformer 简介}
\begin{itemize}
    \item \textbf{Transformer 架构的特点}
        \begin{itemize}
            \item 自注意力机制（Self-Attention）
            \item 多头注意力（Multi-Head Attention）
            \item 位置编码（Positional Encoding）
            \item 编码器-解码器结构（Encoder-Decoder）
        \end{itemize}
    \item \textbf{在自然语言处理中的应用}
        \begin{itemize}
            \item 机器翻译（如：BERT-Base, Transformer-XL）
            \item 文本生成（如：GPT系列）
            \item 文本分类与情感分析
            \item 语音识别与合成
        \end{itemize}
\end{itemize}
\end{frame}
